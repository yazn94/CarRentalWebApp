


- mappings
mysql : mysql 
mongo : mongo 
authentication: auth 
carRental: car-rental 
analytics : analytics 
showResutls : show  



- Running all 
   docker run --name mysql -d -e MYSQL_ROOT_PASSWORD=123456 -p 3306:3306 mysql
   docker run --name mongo -d -p 27017:27017 mongo  
   docker run --name auth -d -p 8080:8080 auth  
   docker run --name analytics -d -p 9500:9500 --link mysql:mysql --link mongo:mongo analytics  
   docker run -d -p 9000:9000 --name car-rental --link mysql:mysql --link auth:auth --link analytics:analytics car-rental 
   docker run -d -p 9999:9999 --name show --link auth:auth --link mysql:myslq --link mongo:mongo show 


- Stopping all 
docker stop mysql
docker stop mongo
docker stop auth
docker stop analytics
docker stop car-rental
docker stop show 


- Removing all  
docker rm -f $(docker ps -aq)
